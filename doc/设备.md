# FabSim 设备配置系统设计方案

## 1. 核心概念

### 1.1 设备类型 vs 设备实例
- **设备类型 (MachineType)**: 可复用的模板，定义设备的基本属性
- **设备实例 (MachineInstance)**: 产线中具体的设备，继承自某个类型，可覆盖参数

### 1.2 配置层级
```
用户自定义类型 (user_machine_types.json)
    ↓ 继承/引用
内置类型 (machine_types.json)
    ↓ 实例化
产线布局中的设备实例 (default_layout.json)
```

## 4. 关键功能实现

### 4.1 设备类型管理器 (MachineTypeManager)

```gdscript
class_name MachineTypeManager
extends RefCounted

# 单例
static var instance: MachineTypeManager

# 内置类型 (只读)
var builtin_types: Dictionary = {}

# 用户自定义类型 (可读写)
var custom_types: Dictionary = {}

# 运行时合并后的类型 (内置 + 自定义)
var all_types: Dictionary = {}

func _init():
    instance = self
    _load_builtin_types()
    _load_custom_types()
    _merge_types()

func _load_builtin_types():
    # 从 res://data/machine_types.json 加载
    pass

func _load_custom_types():
    # 从 user://config/machine_types.json 加载
    pass

func get_type(type_id: String) -> MachineType:
    return all_types.get(type_id)

func save_custom_type(type_data: MachineType) -> bool:
    # 保存到用户配置
    custom_types[type_data.id] = type_data
    return _save_custom_types_to_disk()

func delete_custom_type(type_id: String) -> bool:
    if custom_types.has(type_id):
        custom_types.erase(type_id)
        return _save_custom_types_to_disk()
    return false
```

### 4.2 设备类型数据结构

```gdscript
class_name MachineType
extends RefCounted

var id: String
var base_type: String = ""          # 继承的基础类型ID
var name: String
var description: String = ""

# 视觉配置
var icon: String = "🔧"
var color: String = "#808080"
var sprite_path: String = ""        # 自定义图片路径
var width: int = 100
var height: int = 60
var label_position: String = "bottom"

# 参数定义 (参数名 -> ParameterDef)
var parameters: Dictionary = {}

# 运行时参数
var runtime_config: RuntimeConfig

# MES接口配置
var mes_config: MESConfig

func get_display_name() -> String:
    return name if name else id

func get_effective_parameter(param_name: String):
    # 如果有基础类型，先获取基础类型的值
    if base_type:
        var base = MachineTypeManager.instance.get_type(base_type)
        if base and base.parameters.has(param_name):
            return parameters.get(param_name, base.parameters[param_name])
    return parameters.get(param_name)
```

### 4.3 参数定义系统

```gdscript
class_name ParameterDef
extends RefCounted

enum Type {
    INTEGER,    # 整数
    FLOAT,      # 浮点数
    STRING,     # 字符串
    BOOLEAN,    # 布尔
    ENUM,       # 枚举
    RANGE       # 范围值
}

var name: String                    # 参数名(英文，用于代码)
var display_name: String            # 显示名(中文)
var type: Type
var value: Variant                  # 当前值
var default_value: Variant          # 默认值
var description: String = ""

# 数值型参数限制
var min_value: Variant = null
var max_value: Variant = null
var step: Variant = null            # 步进值

# 枚举选项 (type == ENUM时使用)
var options: Array[String] = []

# 是否可在实例级编辑
var editable: bool = true

# 单位
var unit: String = ""

func validate(new_value: Variant) -> bool:
    match type:
        Type.INTEGER, Type.FLOAT:
            if min_value != null and new_value < min_value:
                return false
            if max_value != null and new_value > max_value:
                return false
        Type.ENUM:
            return new_value in options
    return true
```

## 6. 与仿真系统集成

### 6.1 运行时设备实例

```gdscript
class_name SimMachine
extends Node2D

var machine_id: String
var machine_type: MachineType

# 实例参数 (覆盖值)
var parameter_overrides: Dictionary = {}

# 运行时状态
var status: Status = Status.IDLE
enum Status { IDLE, RUNNING, BLOCKED, DOWN, MAINTENANCE }

var current_lot: Lot = null
var processing_progress: float = 0.0  # 0.0 ~ 1.0

func get_effective_parameter(param_name: String):
    # 优先使用实例覆盖值
    if parameter_overrides.has(param_name):
        return parameter_overrides[param_name]
    # 否则使用类型定义值
    return machine_type.get_effective_parameter(param_name)

func get_process_time() -> float:
    return get_effective_parameter("process_time")

func _ready():
    # 加载视觉资源
    _setup_visual()
    
func _setup_visual():
    # 使用自定义图片或默认色块
    if machine_type.sprite_path:
        var sprite = Sprite2D.new()
        sprite.texture = load(machine_type.sprite_path)
        add_child(sprite)
    else:
        # 使用颜色块 + 图标
        var bg = ColorRect.new()
        bg.color = Color(machine_type.color)
        bg.size = Vector2(machine_type.width, machine_type.height)
        add_child(bg)
```

### 6.2 设备状态机

```gdscript
# SimMachine.gd 续

func update(delta: float):
    match status:
        Status.IDLE:
            _try_start_processing()
        Status.RUNNING:
            _update_processing(delta)
        Status.BLOCKED:
            _check_downstream()
        Status.DOWN:
            _check_repair(delta)

func _try_start_processing():
    if current_lot == null:
        return
    # 检查下游是否可用
    if _is_downstream_available():
        status = Status.RUNNING
        processing_progress = 0.0
        emit_signal("processing_started", machine_id, current_lot.lot_id)
    else:
        status = Status.BLOCKED

func _update_processing(delta: float):
    var process_time = get_process_time()
    processing_progress += delta / (process_time * 60)  # 转换为秒
    
    if processing_progress >= 1.0:
        _complete_processing()

func _complete_processing():
    emit_signal("processing_completed", machine_id, current_lot.lot_id)
    # 转移到下游
    _transfer_to_downstream()
    status = Status.IDLE
    current_lot = null
```

## 7. 界面交互流程

### 7.1 创建设备类型流程

```
1. 用户点击 [+ 新建类型]
   ↓
2. 弹出"新建设备类型"对话框
   - 输入设备ID (自动生成建议值)
   - 选择是否继承现有类型
   - 输入显示名称
   ↓
3. 进入设备类型编辑器
   - 配置视觉属性 (图标、颜色、图片)
   - 配置加工参数
   - 配置运行参数
   - 配置MES接口
   ↓
4. 点击 [保存]
   - 验证数据完整性
   - 保存到 user://config/machine_types.json
   - 刷新类型列表
   - 可在产线编辑器中使用
```

### 7.2 在产线中放置设备

```
1. 从左侧工具箱拖拽/点击设备类型
   ↓
2. 鼠标变成放置模式，显示设备预览
   ↓
3. 在画布上点击放置
   - 自动吸附网格
   - 自动生成设备ID (M1, M2...)
   ↓
4. 自动选中该设备，右侧面板显示属性
   ↓
5. 用户可修改:
   - 位置 (拖拽或输入坐标)
   - 设备名称 (覆盖默认)
   - 参数覆盖值
   - 实例专属配置
```
